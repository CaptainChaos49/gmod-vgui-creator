local upthrust = 10

function ENT:Initialize()
	self = self.Entity
	self:SetModel("models/Combine_Helicopter/helicopter_bomb01.mdl")
	self:SetPos(self:GetPos()-Vector(0,0,60))
	self:PhysicsInitSphere(8)
	self:GetPhysicsObject():Wake()
end

function ENT:Think()
	self:GetPhysicsObject():ApplyForceOffset(Vector(0,0,upthrust),self:GetPos()+Vector(50,0,0))
end
